91 auto IntersectPt(const LineSeg<U>& l) const
92 {
93 auto a1 = p2.y - p1.y;
94 auto b1 = p1.x - p2.x;
95 auto c1 = a1 * (p1.x) + b1 * (p1.y);
96
97 auto a2 = l.p2.y - l.p1.y;
98 auto b2 = l.p1.x - l.p2.x;
99 auto c2 = a2 * (l.p1.x) + b2 * (l.p1.y);
100
101 auto determinant = a1 * b2 - a2 * b1;
102
103 if (determinant == 0.0)
104 {
105 return Vec2<decltype(std::declval<T&>() + std::declval<U&>())>(
106 std::numeric_limits<decltype(std::declval<T&>() + std::declval<U&>())>::max(),
107 std::numeric_limits<decltype(std::declval<T&>() + std::declval<U&>())>::max());
108 }
109 else
110 {
111 auto x = (b2 * c1 - b1 * c2) / determinant;
112 auto y = (a1 * c2 - a2 * c1) / determinant;
113 return Vec2<decltype(std::declval<T&>() + std::declval<U&>())>(x, y);
114 }

139 px = 2 * dy1 - dx1; py = 2 * dx1 - dy1;
140
141 if (dy1 <= dx1)
142 {
143 if (dx >= 0)
144 {
145 x = x1; y = y1; xe = x2;

159 else
160 {

162 px = px + 2 * (dy1 - dx1);

164 out.emplace_back(x, y);
165 }
166 }
