409 template <typename TComp, typename... TArgs>
410 static Component<TComp>& AddComponent(EntityID eid, TArgs&&... args)
411 {
412 ComponentID cid = GetComponentID<TComp>();
413 std::vector<std::unique_ptr<Component<TComp>>>& component_vec = GetComponentVec<TComp>();
414 size_t i = 0;
415
416 // Re-construct component if entity already has the component
417 if (HasComponent(eid, cid))
418 {
419 component_vec.at(entities.at(eid).component_indices.at(cid)) = std::make_unique<Component<TComp>>(TComp(std::forward<TArgs>(args)...), eid);
420 return GetComponent<TComp>(eid);
421 }
422
423 // Add component to component_vec
424 if (destroyed_components.at(cid).empty())
425 {
426 i = component_vec.size();
427 component_vec.emplace_back(new Component<TComp>(TComp(std::forward<TArgs>(args)...), eid));
428 on_component_removals.at(cid).emplace_back(std::vector<std::function<void()>>());
429 }
430 else
431 {
432 auto& available_slots = destroyed_components.at(cid);
433 i = available_slots.back();
434 component_vec.at(i) = std::make_unique<Component<TComp>>(TComp(std::forward<TArgs>(args)...), eid);
435 available_slots.pop_back();
436 }
437
438 // Store component index and field in entity
439 entities.at(eid).component_indices.emplace(cid, i);
440 entities.at(eid).component_field.set(cid);
441 return *component_vec.at(i);
442 }
