828 Mat& GaussianEliminate() &
829 {
830 for (size_t i = 0; i < std::min(row_count, col_count); ++i)
831 {
832 if (At(i, i) == 0)
833 {
834 for (size_t j = 1; j < row_count - i; ++j)
835 {
836 if (At(i + j, i) != 0)
837 {
838 SwapRows(i, i + j);
839 break;
840 }
841 }
842 }
843
844 for (size_t j = 0; j < row_count; ++j)
845 {
846 if (At(i, i) == 0 || j == i)
847 {
848 continue;
849 }
850 AddRow(j, i, -At(j, i) / At(i, i));
851 }
852
853 if (At(i, i) != 0)
854 MultiplyRow(i, 1 / At(i, i));
855 }
856
857 return *this;

876 template <typename U = T>
877 U Determinant() const
878 {
879 if (row_count != col_count)
880 throw MatNotSq<T>(row_count, col_count);
881
882 U det = 0;
883
884 for (size_t j = 0; j < col_count; ++j)
885 {
886 U pos_diag = 1;
887 U neg_diag = 1;
888
889 for (size_t i = 0; i < row_count; ++i)
890 {
891 pos_diag *= At(i, (j + i) % col_count);
892 neg_diag *= At(row_count - i - 1, (j + i) % col_count);
893 }
894
895 det += pos_diag - neg_diag;
896 }
897
898 return det;
