367 template <typename TComp, typename... TArgs>
368 static Component<TComp>& AddComponent(EntityID eid, TArgs&&... args)
369 {
370 ComponentID cid = GetComponentID<TComp>();
371 std::vector<std::unique_ptr<Component<TComp>>>& component_vec = GetComponentVec<TComp>();
372 size_t i = 0;
373
374 // Re-construct component if entity already has the component
375 if (HasComponent(eid, cid))
376 {
377 component_vec.at(entities.at(eid).component_indices.at(cid)) = std::make_unique<Component<TComp>>(TComp(std::forward<TArgs>(args)...), eid);
378 return GetComponent<TComp>(eid);
379 }
380
381 // Add component to component_vec
382 if (destroyed_components.at(cid).empty())
383 {
384 i = component_vec.size();
385 component_vec.emplace_back(new Component<TComp>(TComp(std::forward<TArgs>(args)...), eid));
386 on_component_removals.at(cid).emplace_back(std::vector<std::function<void()>>());
387 }
388 else
389 {
390 auto& available_slots = destroyed_components.at(cid);
391 i = available_slots.back();
392 component_vec.at(i) = std::make_unique<Component<TComp>>(TComp(std::forward<TArgs>(args)...), eid);
393 available_slots.pop_back();
394 }
395
396 // Store component index and field in entity
397 entities.at(eid).component_indices.emplace(cid, i);
398 entities.at(eid).component_field.set(cid);
399 return *component_vec.at(i);
400 }
